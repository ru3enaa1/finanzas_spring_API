package com.pfm.dao;

import com.pfm.model.Usuario;
import com.pfm.util.DBConnection;

import java.nio.charset.StandardCharsets;
import java.security.MessageDigest;
import java.sql.*;

public class UsuarioDAO {

    // --- Utilities ---
    private String sha256(String input) throws Exception {
        MessageDigest md = MessageDigest.getInstance("SHA-256");
        byte[] hashed = md.digest(input.getBytes(StandardCharsets.UTF_8));
        StringBuilder sb = new StringBuilder();
        for (byte b : hashed) sb.append(String.format("%02x", b));
        return sb.toString();
    }

    // --- CRUD & auth ---

    public boolean registrar(Usuario u) {
        String sql = "INSERT INTO usuario (nombre, apellido, correo, contrasena) VALUES (?, ?, ?, ?)";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setString(1, u.getNombre());
            ps.setString(2, u.getApellido());
            ps.setString(3, u.getCorreo());
            ps.setString(4, sha256(u.getContrasena())); // store hash
            return ps.executeUpdate() > 0;
        } catch (Exception e) {
            e.printStackTrace(); return false;
        }
    }

    /** Returns the full user if login is valid, else null. */
    public Usuario login(String correo, String contrasena) {
        String sql = "SELECT * FROM usuario WHERE correo=?";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setString(1, correo);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) {
                    String storedHash = rs.getString("contrasena");
                    if (storedHash != null && storedHash.equals(sha256(contrasena))) {
                        Usuario u = new Usuario();
                        u.setId(rs.getInt("id"));
                        u.setNombre(rs.getString("nombre"));
                        u.setApellido(rs.getString("apellido"));
                        u.setCorreo(rs.getString("correo"));
                        u.setContrasena(storedHash);
                        return u;
                    }
                }
            }
        } catch (Exception e) { e.printStackTrace(); }
        return null;
    }

package com.pfm.dao;

import com.pfm.model.Transaccion;
import com.pfm.util.DBConnection;

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class TransaccionDAO {

    public boolean crear(Transaccion t) {
        if (t.getCuentaId() == null) {
            throw new IllegalArgumentException("cuenta_id es obligatorio para registrar la transacción (RF12).");
        }
        String sql = "INSERT INTO transaccion (cuenta_id, tipo, categoria, fecha, monto, descripcion, fijo) " +
                     "VALUES (?, ?, ?, ?, ?, ?, ?)";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {
            ps.setInt(1, t.getCuentaId());
            ps.setString(2, t.getTipo());
            ps.setString(3, t.getCategoria());
            ps.setDate(4, t.getFecha());
            ps.setDouble(5, t.getMonto());
            ps.setString(6, t.getDescripcion());
            ps.setBoolean(7, t.isFijo());
            return ps.executeUpdate() > 0;
        } catch (Exception e) { e.printStackTrace(); return false; }
    }

    public Transaccion obtener(int id, int usuarioId) {
        String sql =
            "SELECT t.* FROM transaccion t " +
            "JOIN cuenta c ON t.cuenta_id=c.id " + // use LEFT JOIN if cuenta_id puede ser NULL
            "WHERE t.id=? AND c.usuario_id=?";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setInt(1, id);
            ps.setInt(2, usuarioId);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) {
                    Transaccion t = new Transaccion();
                    t.setId(rs.getInt("id"));
                    t.setCuentaId((Integer) rs.getObject("cuenta_id"));
                    t.setTipo(rs.getString("tipo"));
                    t.setCategoria(rs.getString("categoria"));
                    t.setFecha(rs.getDate("fecha"));
                    t.setMonto(rs.getDouble("monto"));
                    t.setDescripcion(rs.getString("descripcion"));
                    t.setFijo(rs.getBoolean("fijo"));
                    return t;
                }
            }
        } catch (Exception e) { e.printStackTrace(); }
        return null;
    }

    public boolean actualizar(Transaccion t, int usuarioId) {
        String sql =
            "UPDATE transaccion t " +
            "JOIN cuenta c ON t.cuenta_id=c.id " + // LEFT JOIN if nullable
            "SET t.tipo=?, t.categoria=?, t.fecha=?, t.monto=?, t.descripcion=?, t.fijo=? " +
            "WHERE t.id=? AND c.usuario_id=?";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setString(1, t.getTipo());
            ps.setString(2, t.getCategoria());
            ps.setDate(3, t.getFecha());
            ps.setDouble(4, t.getMonto());
            ps.setString(5, t.getDescripcion());
            ps.setBoolean(6, t.isFijo());
            ps.setInt(7, t.getId());
            ps.setInt(8, usuarioId);
            return ps.executeUpdate() > 0;
        } catch (Exception e) { e.printStackTrace(); return false; }
    }

    public boolean eliminar(int id, int usuarioId) {
        String sql =
            "DELETE t FROM transaccion t " +
            "JOIN cuenta c ON t.cuenta_id=c.id " + // LEFT JOIN if nullable
            "WHERE t.id=? AND c.usuario_id=?";
        try (Connection c = DBConnection.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setInt(1, id);
            ps.setInt(2, usuarioId);
            return ps.executeUpdate() > 0;
        } catch (Exception e) { e.printStackTrace(); return false; }
    }

    public List<Transaccion> listarPorUsuarioMes(int usuarioId, Integer anio, Integer mes) {
        List<Transaccion> list = new ArrayList<>();
        StringBuilder sb = new StringBuilder(
            "SELECT t.* FROM transaccion t JOIN cuenta c ON t.cuenta_id=c.id WHERE c.usuario_id=?");
        // If nullable cuenta_id is allowed, change to LEFT JOIN to avoid hiding rows
        if (anio != null && mes != null) sb.append(" AND YEAR(t.fecha)=? AND MONTH(t.fecha)=?");

package com.pfm.servlets;

import com.pfm.dao.UsuarioDAO;
import com.pfm.model.Usuario;
import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import java.io.IOException;

@WebServlet("/auth")
public class AuthServlet extends HttpServlet {
    private final UsuarioDAO usuarioDAO = new UsuarioDAO();

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String action = req.getParameter("action");

        if ("register".equalsIgnoreCase(String.valueOf(action))) {
            Usuario u = new Usuario();
            u.setNombre(req.getParameter("nombre"));
            u.setApellido(req.getParameter("apellido"));
            u.setCorreo(req.getParameter("correo"));
            u.setContrasena(req.getParameter("contrasena"));
            if (usuarioDAO.registrar(u)) {
                resp.sendRedirect("login.jsp");
            } else {
                req.setAttribute("error", "No fue posible registrar el usuario (correo ya existe?).");
                req.getRequestDispatcher("register.jsp").forward(req, resp);
            }
            return;
        }

        // default → login flow (covers null/empty action)
        String correo = req.getParameter("correo");
        String contrasena = req.getParameter("contrasena");
        Usuario u = usuarioDAO.login(correo, contrasena);
        if (u != null) {
            req.getSession().setAttribute("usuario", u);
            resp.sendRedirect("dashboard.jsp");
        } else {
            req.setAttribute("error", "Credenciales inválidas");
            req.getRequestDispatcher("login.jsp").forward(req, resp);
        }
    }
}

    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        String action = req.getParameter("action");
        if ("logout".equals(action)) {
            req.getSession().invalidate();
            resp.sendRedirect("login.jsp");
        }
    }
}

        Usuario u = (Usuario) req.getSession().getAttribute("usuario");
        if (u == null) { resp.sendRedirect("login.jsp"); return; }
        List<Cuenta> cuentas = cuentaDAO.listarPorUsuario(u.getId());
        req.setAttribute("cuentas", cuentas);
req.getRequestDispatcher("creacion-cuentas.jsp").forward(req, resp);

    }

    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        Usuario u = (Usuario) req.getSession().getAttribute("usuario");
        if (u == null) { resp.sendRedirect("login.jsp"); return; }
        String action = req.getParameter("action");
        if ("crear".equals(action)) {
            Cuenta c = new Cuenta();
            c.setUsuarioId(u.getId());
            c.setTipo(req.getParameter("tipo"));
            c.setSaldo(Double.parseDouble(req.getParameter("saldo")));
            cuentaDAO.crear(c);

            double aportes  = fondoDAO.totalAportesUsuarioMes(usuarioId, anio, mes);
            double ahorro   = ingresos - gastos - aportes;

            sb.append("Totales del mes:\n");
            sb.append("Ingresos: ").append(ingresos).append("\n");
            sb.append("Gastos  : ").append(gastos).append("\n");
            sb.append("Aportes : ").append(aportes).append("\n");
            sb.append("Ahorro  : ").append(ahorro).append("\n");
        } else {
            // Informe anual: acumular a partir de la BD
            try (Connection c = DBConnection.getConnection();
                 PreparedStatement ps = c.prepareStatement(
                     "SELECT MONTH(t.fecha) m, " +
                     "SUM(CASE WHEN t.tipo='Ingreso' THEN t.monto ELSE 0 END) ingresos, " +
                     "SUM(CASE WHEN t.tipo='Gasto'   THEN t.monto ELSE 0 END) gastos " +
                     "FROM transaccion t JOIN cuenta c ON t.cuenta_id=c.id " +
                     "WHERE c.usuario_id=? AND YEAR(t.fecha)=? " +
                     "GROUP BY MONTH(t.fecha) ORDER BY m")) {
                ps.setInt(1, usuarioId);
                ps.setInt(2, anio);
                try (ResultSet rs = ps.executeQuery()) {
                    sb.append("Resumen anual por mes:\n");
                    while (rs.next()) {
                        int m = rs.getInt("m");
                        double ingresos = rs.getDouble("ingresos");
                        double gastos   = rs.getDouble("gastos");
                        double aportes  = fondoDAO.totalAportadoMesUsuario(usuarioId, anio, m);
                        double ahorro   = ingresos - gastos - aportes;
                        sb.append(String.format("Mes %02d  Ingresos: %.2f  Gastos: %.2f  Aportes: %.2f  Ahorro: %.2f%n",
                                                m, ingresos, gastos, aportes, ahorro));
                    }
                }
            } catch (Exception e) {
                sb.append("Error generando informe anual: ").append(e.getMessage()).append("\n");
            }
        }
        return sb.toString();
    }

    private void guardarInforme(int usuarioId, String tipo, String contenido) {
    final String sql = "INSERT INTO informe (usuario_id, tipo, contenido, fecha_generacion) VALUES (?, ?, ?, ?)";
    try (Connection c = DBConnection.getConnection();
         PreparedStatement ps = c.prepareStatement(sql)) {
        ps.setInt(1, usuarioId);
        ps.setString(2, tipo); // "Mensual" or "Anual"
        ps.setString(3, contenido);
        ps.setDate(4, java.sql.Date.valueOf(java.time.LocalDate.now())); // explicit value
        ps.executeUpdate();
    } catch (Exception e) {
        e.printStackTrace();
    }
}

    /** Exporta el contenido como archivo CSV muy simple (texto de una sola columna). */
    private void exportarCSV(HttpServletResponse resp, String tipo, int anio, Integer mes, String contenido) throws IOException {
        String fileName = "informe_" + tipo.toLowerCase() + "_" + anio + (mes != null ? "_" + mes : "") + ".csv";
        resp.setContentType("text/csv; charset=UTF-8");
        resp.setHeader("Content-Disposition", "attachment; filename=\"" + fileName + "\"");
        try (PrintWriter out = resp.getWriter()) {
            out.println("contenido");
            for (String line : contenido.split("\\R")) {
                out.print('"');
                out.print(line.replace("\"", "\"\""));
                out.println('"');
            }
        }
    }
}

README.md

1) Resumen

Aplicación web de finanzas personales con Java Servlets + JSP + JDBC (MySQL). Permite gestionar usuarios, cuentas, transacciones (ingresos/gastos), presupuestos, gastos fijos, fondos/bolsas de ahorro, alertas y generar informes (mensual/anual) con exportación CSV.

Este README acompaña el proyecto para su evaluación académica (“teacher-ready”): instalación, ejecución en Tomcat, estructura, y trazabilidad de requerimientos.

2) Requisitos previos

Java 11+

Maven 3.8+

MySQL 8+

Apache Tomcat 9/10 (o compatible con Servlet 5 API en provided)

Editor/IDE (Eclipse, IntelliJ o VS Code + Extensiones)

3) Instalación de la base de datos

Cree una base de datos MySQL: CREATE DATABASE finanzas_personales;

Importe el script: database/finanzas_personales.sql.
El script incluye tablas para usuario, cuenta, transaccion, presupuesto, gasto_fijo, registro_gasto_fijo, fondo, registro_fondo, resumen_mensual, alerta, informe.

Verifique credenciales en src/main/java/com/pfm/util/DBConnection.java y ajústelas a su entorno local (URL, USER, PASS).

4) Configuración del proyecto

Estructura (resumen):

pfm/
├─ pom.xml
├─ src/main/java/com/pfm/...
│  ├─ util/DBConnection.java
│  ├─ model/ (Usuario, Cuenta, Transaccion, Presupuesto, GastoFijo, RegistroGastoFijo, Fondo, RegistroFondo, ResumenMensual, Alerta)
│  ├─ dao/ (UsuarioDAO, CuentaDAO, TransaccionDAO, PresupuestoDAO, GastoFijoDAO, FondoDAO, ResumenDAO, AlertaDAO)
│  └─ servlets/ (AuthServlet, CuentaServlet, TransaccionServlet, PresupuestoServlet, GastoFijoServlet, FondoServlet, ResumenServlet, AlertaServlet, InformeServlet)
└─ src/main/webapp/
   ├─ WEB-INF/web.xml
   ├─ *.jsp (index, login, register, dashboard, cuentas, transacciones, presupuestos, alertas, informes, fondos, gastos-fijos)
   └─ assets CSS/JS (reusar los .html/.css/.js dados)


web.xml declara EncodingFilter (UTF-8).

pom.xml empaqueta WAR y trae JSTL + MySQL JDBC.

5) Compilación y despliegue

Compilar: mvn clean package

Desplegar el pfm.war en Tomcat:

Copiar target/pfm.war a TOMCAT_HOME/webapps/

Iniciar Tomcat y abrir http://localhost:8080/pfm/

Codificación: Se fuerza UTF-8 por filtro.

6) Credenciales y sesión

Regístrese desde register.jsp y luego inicie sesión (login.jsp). La sesión se valida en cada servlet/JSP, redirigiendo a login.jsp si no hay usuario autenticado (ver servlets).

7) Funcionalidad (flujo por secciones)
7.1 Cuentas y Transacciones

Cuentas: alta/listado/ajuste de saldo (DAO + CuentaServlet).

Transacciones (ingresos/gastos variables) sobre cuentas, recalculando saldo inmediatamente (TransaccionServlet).

Los formularios están en transacciones.jsp y se listan por año/mes. Al registrar gastos, se valida presupuesto y se emite alerta si se excede.

7.2 Presupuestos (por categoría y mes)

CRUD básico vía PresupuestoServlet + PresupuestoDAO.

Comparación real vs. planificado y alertas al exceder (ver integración con transacciones).

7.3 Gastos fijos

Gestión de gastos fijos y su pago mensual en GastoFijoServlet + GastoFijoDAO (tabla registro_gasto_fijo).

Genera alerta si un fijo del mes queda no pagado. (UI en gastos-fijos.jsp.)

7.4 Fondos / Bolsas de ahorro

Definición de fondos con meta anual y aportes mensuales (FondoServlet + FondoDAO y registro_fondo).

Cálculo de avance (%) y alertas cuando se alcanzan umbrales/meta. (UI en fondos.jsp.)

7.5 Resumen mensual

ResumenServlet calcula ingresos, gastos, aportes y ahorro neto y almacena en resumen_mensual mediante ResumenDAO.upsert(...). La vista indica Ahorro o Déficit.

7.6 Alertas

AlertaServlet lista, elimina individuales o “limpia todas”. UI en alertas.jsp.

7.7 Informes (Mensual / Anual) + Exportación CSV

InformeServlet genera contenido (mensual: totales y cumplimiento de presupuestos; anual: consolidado por mes), guarda en informe y permite descargar CSV. UI en informes.jsp.

8) Mapeo de Requerimientos → Implementación

(Extracto de RFs relevantes a esta entrega)

RF1–RF5 (Usuarios/Cuentas/Saldo): Servlets y DAOs correspondientes; saldo se actualiza al registrar transacciones.

RF6–RF9 (Gastos fijos + alertas por impago): GastoFijoServlet/GastoFijoDAO + registro_gasto_fijo. Alerta si no se paga el mes.

RF10–RF13 (Transacciones + recálculo de saldo): TransaccionServlet/TransaccionDAO + CuentaDAO.ajustarSaldo(...).

RF14–RF17 (Fondos + % avance + alertas): FondoServlet/FondoDAO + registro_fondo.

RF18–RF20 (Resumen mensual + almacenamiento + estado ahorro/déficit): ResumenServlet + ResumenDAO/resumen_mensual.

RF21–RF23 (Presupuestos + comparación + alerta por exceso): PresupuestoServlet/PresupuestoDAO + ver lógica en TransaccionServlet.

RF24–RF25 (Alertas y visualización): AlertaDAO + AlertaServlet + alertas.jsp.

RF26–RF28 (Informes + exportación + consulta): InformeServlet + informes.jsp.

9) Endpoints (principalmente por anotación @WebServlet)

/auth (login/logout/registro)

/cuentas, /transacciones, /presupuestos, /gastos-fijos, /fondos, /alertas, /informes
(Cada uno con GET para vistas/listados y POST para acciones de creación/actualización, según el caso).

10) Datos de prueba

Cree un usuario desde register.jsp.

Cree cuentas y registre ingresos/gastos en “Transacciones”.

Defina presupuestos por categoría/mes.

Registre gastos fijos y su pago mensual.

Cree fondos y realice aportes por mes.

Genere informes y pruebe CSV.

11) Consideraciones de despliegue

Si usa Jakarta Servlet 6.0 (and JSTL Jakarta 3.0.1), lo cual mapea Tomcat 10.1.x+ and Java 17 en pom.xml Compilar con Maven y desplegar el WAR.

Verificar DBConnection y firewall/puerto MySQL.

12) Entrega (consigna académica)

Entregar carpeta comprimida con todo el proyecto + enlace al repositorio (versionamiento). Siga el formato de nombre solicitado.
